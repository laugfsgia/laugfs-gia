Permissions:
How this works:
Roles define what each group of users can do. The built-in admin role cannot be deleted or restricted — it always has full access.

For each custom role, toggle the permissions it should have. Users inherit exactly the permissions of their assigned role.
Available permissions:

Connect to a single Google Sheet named LAUGFS GIA. Each financial year is saved to its own tab automatically. If you connected before, you must replace the script and redeploy (new version) — the script has changed.

  1. Open your LAUGFS GIA Google Sheet
  2. Click Extensions → Apps Script
  3. Delete all existing code, paste the script below
  4. Click Deploy → Manage deployments → ✏ Edit → New version → Deploy
    (If first time: Deploy → New deployment → Web app → Execute as: Me → Anyone → Deploy)
  5. Copy the Web app URL, paste below and click Save URL
  6. Click ↑ Sync All Records Now to push existing data

function doGet(e){
  var wb = SpreadsheetApp.getActiveSpreadsheet();
  var p  = e.parameter;
  var action = p.action || 'add';

  // GET CONFIG — returns JSON, called on startup by every user
  if(action === 'get-config'){
    var cs = wb.getSheetByName('Config');
    var cfg = {};
    if(cs && cs.getLastRow() > 1){
      cs.getDataRange().getValues().slice(1).forEach(function(r){ if(r[0]) cfg[r[0]]=r[1]; });
    }
    return ContentService.createTextOutput(JSON.stringify(cfg))
      .setMimeType(ContentService.MimeType.JSON);
  }

  // SAVE CONFIG — called when admin changes any lookup
  if(action === 'save-config'){
    var cs = wb.getSheetByName('Config');
    if(!cs){
      cs = wb.insertSheet('Config');
      cs.appendRow(['Key','Value','Last Updated']);
      cs.getRange(1,1,1,3).setFontWeight('bold');
      cs.setFrozenRows(1);
    }
    var keys=['pillars','subsegs','ratings','companies','statuses','followups','reviewers','auditors','roles','labels','users'];
    var existingRows={};
    if(cs.getLastRow()>1) cs.getDataRange().getValues().slice(1).forEach(function(r,i){ if(r[0]) existingRows[r[0]]=i+2; });
    keys.forEach(function(k){
      if(!p[k]) return;
      var ts=new Date().toISOString();
      if(existingRows[k]) cs.getRange(existingRows[k],2,1,2).setValues([[p[k],ts]]);
      else cs.appendRow([k,p[k],ts]);
    });
  }

  // ADD / UPDATE AUDIT
  if(action === 'add' || action === 'update'){
    var sheetName = p.fy || 'Unassigned';
    var s = wb.getSheetByName(sheetName);
    if(!s){
      s = wb.insertSheet(sheetName);
      s.appendRow(['Ref #','FY','Company','Pillar','Sub-segment','Status',
        'Senior Auditor','Aud 01','Aud 02','Aud 03','Aud 04','Aud 05',
        'Start','End','1st Review','2nd Review','AR Issued',
        '1st Reviewer','2nd Reviewer','Planned Hrs','Actual Hrs',
        'Travelling (LKR)','Food & Subs (LKR)','Accommodation (LKR)',
        'Other Cost (LKR)','Other Desc','Total Cost (LKR)',
        'Remarks','Created By','Synced At']);
      s.getRange(1,1,1,30).setFontWeight('bold');
      s.setFrozenRows(1);
    }
    if(action === 'update'){
      var data = s.getDataRange().getValues();
      for(var i = data.length-1; i >= 1; i--){
        if(data[i][0] === p.ref){ s.deleteRow(i+1); break; }
      }
    }
    s.appendRow([p.ref,p.fy,p.company,p.pillar,p.subseg,p.status,
      p.senior||'',p.aud1||'',p.aud2||'',p.aud3||'',p.aud4||'',p.aud5||'',
      p.start||'',p.end||'',p.rev1||'',p.rev2||'',p.ar||'',
      p.reviewer||'',p.reviewer2||'',
      parseFloat(p.plannedHrs)||0,parseFloat(p.actualHrs)||0,
      parseFloat(p.costTravel)||0,parseFloat(p.costSubs)||0,
      parseFloat(p.costAccom)||0,parseFloat(p.costOther)||0,
      p.costOtherDesc||'',parseFloat(p.costTotal)||0,
      p.remarks||'',p.createdBy||'',new Date().toISOString()]);
  }

  // DELETE AUDIT — removes from FY tab, archives to Deleted Audits
  if(action === 'delete'){
    var sheetName = p.fy || 'Unassigned';
    var s = wb.getSheetByName(sheetName);
    var deletedRow = null;
    if(s){
      var data = s.getDataRange().getValues();
      for(var i = data.length-1; i >= 1; i--){
        if(data[i][0] === p.ref){ deletedRow = data[i]; s.deleteRow(i+1); break; }
      }
    }
    var del = wb.getSheetByName('Deleted Audits');
    if(!del){
      del = wb.insertSheet('Deleted Audits');
      del.appendRow(['Ref #','FY','Company','Pillar','Sub-segment','Status',
        'Senior Auditor','Aud 01','Aud 02','Aud 03','Aud 04','Aud 05',
        'Start','End','1st Review','2nd Review','AR Issued',
        '1st Reviewer','2nd Reviewer','Planned Hrs','Actual Hrs',
        'Travelling (LKR)','Food & Subs (LKR)','Accommodation (LKR)',
        'Other Cost (LKR)','Other Desc','Total Cost (LKR)',
        'Remarks','Created By','Synced At','Deleted At']);
      del.getRange(1,1,1,31).setFontWeight('bold');
      del.getRange(1,1,1,31).setBackground('#fce8e6');
      del.setFrozenRows(1);
    }
    if(deletedRow){ del.appendRow(deletedRow.concat([new Date().toISOString()])); }
    else { del.appendRow([p.ref,p.fy,p.company||'',p.pillar||'',p.subseg||'',p.status||'',
      '','','','','','','','','','','','','','','','','','','','','','','','',
      new Date().toISOString()]); }
  }

  // CLEAR ALL — wipes data rows from FY tabs only, preserves Config and Deleted Audits
  if(action === 'clear-all'){
    var skip=['Deleted Audits','Config'];
    wb.getSheets().forEach(function(s){
      if(skip.indexOf(s.getName())>=0) return;
      var last=s.getLastRow(); if(last>1) s.deleteRows(2,last-1);
    });
  }

  // ── TIMESHEET — ADD ──────────────────────────────────────────────────────
  if(action === 'add-timesheet' || action === 'update-timesheet'){
    var ts = wb.getSheetByName('Timesheet');
    if(!ts){
      ts = wb.insertSheet('Timesheet');
      ts.appendRow(['ID','Date','Auditor','Activity Type','Location','Hours','Linked Audit Ref','Remarks','Created By','Synced At']);
      ts.getRange(1,1,1,10).setFontWeight('bold');
      ts.getRange(1,1,1,10).setBackground('#E8F0FE');
      ts.setFrozenRows(1);
    }
    if(action === 'update-timesheet'){
      var tsData = ts.getDataRange().getValues();
      for(var i = tsData.length-1; i >= 1; i--){
        if(tsData[i][0] === p.tsId){ ts.deleteRow(i+1); break; }
      }
    }
    ts.appendRow([p.tsId||'',p.date||'',p.auditor||'',p.activity||'',p.location||'',
      parseFloat(p.hours)||0, p.auditRef||'', p.remarks||'', p.createdBy||'', new Date().toISOString()]);
  }

  // ── TIMESHEET — DELETE ────────────────────────────────────────────────────
  if(action === 'delete-timesheet'){
    var ts = wb.getSheetByName('Timesheet');
    if(ts){
      var tsData = ts.getDataRange().getValues();
      for(var i = tsData.length-1; i >= 1; i--){
        if(tsData[i][0] === p.tsId){ ts.deleteRow(i+1); break; }
      }
    }
  }

  // ── TIMESHEET — CLEAR ALL ─────────────────────────────────────────────────
  if(action === 'clear-timesheet'){
    var ts = wb.getSheetByName('Timesheet');
    if(ts){
      var last = ts.getLastRow();
      if(last > 1) ts.deleteRows(2, last-1);
    }
  }

  return ContentService.createTextOutput('ok')
    .setMimeType(ContentService.MimeType.TEXT);
}
